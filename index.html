while True:
    print("\nBirim Dönüştürücü")
    print("0: çıkış")
    print("1: km -> m")
    print("2: m -> cm")
    print("3: cm -> mm")
    print("4: kg -> g")
    print("5: g -> mg")
    print("6: L -> mL")
    print("7: saat -> dakika")
    print("8: dakika -> saniye")
    print("9: gun -> saat")
    print("10: hafta -> gun")
    print("11: ay -> gun (ortalama)")
    print("12: yil -> gun")
    print("13: km/h -> m/s")
    print("14: °C -> °F")
    print("15: °F -> °C")
    print("16: inc -> cm")
    print("17: feet -> m")
    print("18: mil -> km")
    print("19: pound -> kg")
    print("20: ons -> g")
    print("21: m -> km")
    print("22: cm -> m")

    secim = int(input("Hangi dönüşümü yapmak istersiniz? (0-22): "))
    if secim == 0:
        print("Program kapatılıyor...")
        break

    deger = float(input("Donusturulecek degeri girin: "))

    if secim == 1:
        print(deger, "km =", deger * 1000, "m")
    elif secim == 2:
        print(deger, "m =", deger * 100, "cm")
    elif secim == 3:
        print(deger, "cm =", deger * 10, "mm")
    elif secim == 4:
        print(deger, "kg =", deger * 1000, "g")
    elif secim == 5:
        print(deger, "g =", deger * 1000, "mg")
    elif secim == 6:
        print(deger, "L =", deger * 1000, "mL")
    elif secim == 7:
        print(deger, "saat =", deger * 60, "dakika")
    elif secim == 8:
        print(deger, "dakika =", deger * 60, "saniye")
    elif secim == 9:
        print(deger, "gun =", deger * 24, "saat")
    elif secim == 10:
        print(deger, "hafta =", deger * 7, "gun")
    elif secim == 11:
        print(deger, "ay =", deger * 30, "gun")
    elif secim == 12:
        print(deger, "yil =", deger * 365, "gun")
    elif secim == 13:
        print(deger, "km/h =", deger / 3.6, "m/s")
    elif secim == 14:
        print(deger, "°C =", (deger * 9 / 5) + 32, "°F")
    elif secim == 15:
        print(deger, "°F =", (deger - 32) * 5 / 9, "°C")
    elif secim == 16:
        print(deger, "inc =", deger * 2.54, "cm")
    elif secim == 17:
        print(deger, "feet =", deger * 0.3048, "m")
    elif secim == 18:
        print(deger, "mil =", deger * 1.60934, "km")
    elif secim == 19:
        print(deger, "pound =", deger * 0.453592, "kg")
    elif secim == 20:
        print(deger, "ons =", deger * 28.3495, "g")
    elif secim == 21:
        print(deger, "m =", deger / 1000, "km")
    elif secim == 22:
        print(deger, "cm =", deger / 100, "m")
    else:
        print("Gecersiz secim!")
        continue

    devam = input("Devam etmek istiyor musunuz? (e/h): ")
    if devam == "e":
        continue
    else:
        print("Program kapatildi.")
        break

# 0 yazinca program break ile biter.
# Kullanici 'e' yazarsa continue ile basa doner.
# Bu sekilde hem while, hem break hem de continue komutlarini
